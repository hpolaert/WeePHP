114:   /**
115:    * Send the mail !
116:    *
117:    * @param   boolean $throw Optional parameter to return an exception instead of an error
118:    *
119:    * @return  mixed
120:    * @throws MailException
121:    */
122:   public function send($throw = FALSE) {
123:     // Returns first encountered error
124:     if ($this->_err) {
125:       return FALSE;
126:     }
127:     // CC & BCC recipients
128:     $recipients = implode(',', $this->_recipients);
129:     $ccRecipients = implode(',', $this->_ccRecipients);
130:     $bccRecipients = implode(',', $this->_bccRecipients);
131:     // Must have at least one recipient
132:     if ($recipients === '' && $ccRecipients === '' && $bccRecipients === '') {
133:       // otherwise throw error
134:       $this->throwError(1);
135:     }
136:     // Set the sender of the email
137:     if ($this->_sender !== NULL && $this->_sender !== "") {
138:       ini_set('sendmail_from', $this->_sender);
139:     }
140:     // Return path and reply-to
141:     $returnPath = $this->_returnPath != NULL ? $this->_returnPath : $this->_sender;
142:     $replyTo = $this->_replyTo != NULL ? $this->_returnPath : $this->_sender;
143:     $returnPathDebug = '-f' . $this->_returnPath != NULL ? $this->_returnPath : $this->_sender;
144:     // Content => by default HTML content, if otherwise specified strip html tags for text content (with a basic BR to NL conversion)
145:     // As it's possible to send an email without content, the field is not mandatory
146:     $htmlContent = $this->_bodyContent != NULL ? $this->_bodyContent : ' ';
147:     $textContent = $this->_alternateContent != NULL ? $this->_alternateContent : strip_tags(preg_replace('#<br\s*?/?>#i', PHP_EOL, $this->_bodyContent));
148:     // Defines encoding & email-subject
149:     $encoding = ($this->_utf8 != NULL && $this->_utf8 != FALSE) ? 'utf-8' : 'iso-8859-1';
150:     $subject = $this->_subject != NULL ? $this->_subject : ' ';
151:     if ($encoding === 'utf-8') {
152:       $subject = '=?UTF-8?B?' . base64_encode($subject) . '?=';
153:     }
154:     // Boundary to enable different content-type
155:     $boundary = md5(time());
156:     //################# HEADERS
157:     $headers = (string) NULL;
158:     $headers .= 'From: ' . $this->_sender . PHP_EOL;
159:     $headers .= 'Reply-To: ' . $replyTo . PHP_EOL;
160:     $headers .= 'Return-Path: ' . $returnPath . PHP_EOL;
161:     $headers .= $ccRecipients !== '' ? 'Cc: ' . $ccRecipients . PHP_EOL : '';
162:     $headers .= $bccRecipients !== '' ? 'Bcc: ' . $bccRecipients . PHP_EOL : '';
163:     $headers .= 'Message-ID:<' . time() . '-' . md5($this->_sender) . '@' . $_SERVER['SERVER_NAME'] . '>' . PHP_EOL;
164:     $headers .= 'X-Mailer: PHP v' . phpversion() . PHP_EOL;
165:     $headers .= 'MIME-Version: 1.0' . PHP_EOL;
166:     $headers .= 'Content-Type: multipart/alternative; boundary="' . $boundary . '"' . PHP_EOL;
167:     //################# BODY
168:     $body = (string) NULL;
169:     //################# ATTACHMENTS
170:     if (!empty($this->_attachments)) {
171:       foreach ($this->_attachments as $attachment) {
172:         // Get attachment content and basename
173:         $attachmentBasename = preg_replace('/^.+[\\\\\\/]/', '', $attachment);
174:         $data = chunk_split(base64_encode(file_get_contents($attachment)));
175:         // Add attachment
176:         $body .= '--' . $boundary . PHP_EOL;
177:         $body .= 'Content-Type: ' . $this->getMimeType($attachment) . '; name="' . $attachmentBasename . '"' . PHP_EOL;
178:         $body .= 'Content-Transfer-Encoding: base64' . PHP_EOL;
179:         $body .= 'Content-Disposition: attachment; filename="' . $attachmentBasename . '"' . PHP_EOL . PHP_EOL;
180:         $body .= $data . PHP_EOL . PHP_EOL;
181:       }
182:     }
183:     //################# MAIL CONTENT
184:     $body .= 'Content-Type: multipart/alternative' . PHP_EOL;
185:     // Text content
186:     $body .= '--' . $boundary . PHP_EOL;
187:     $body .= 'Content-Type: text/plain; charset=' . $encoding . PHP_EOL;
188:     $body .= 'Content-Transfer-Encoding: 8bit' . PHP_EOL;
189:     $body .= $textContent . PHP_EOL . PHP_EOL;
190:     // HTML content
191:     $body .= '--' . $boundary . PHP_EOL;
192:     $body .= 'Content-Type: text/html; charset=' . $encoding . PHP_EOL;
193:     $body .= 'Content-Transfer-Encoding: 8bit' . PHP_EOL;
194:     $body .= $htmlContent . PHP_EOL . PHP_EOL;
195:     //################# CLOSING BODY
196:     $body .= '--' . $boundary . '--' . PHP_EOL . PHP_EOL;
197:     // Send and optionally returns technical error
198:     if (!mail($recipients, $subject, $body, $headers, $returnPathDebug)) {
199:       // Throw exception if mail cannot be sent
200:       if ($throw && !$this->_internalErrorHandler) {
201:         $errorMsg = $this->_errors[3][1];
202:         $errorCategory = $this->_errors[3][0];
203:         throw new MailException($errorCategory, $errorMsg, 3);
204:       }
205:       return FALSE;
206:     } else {
207:       // Reset INI sender
208:       ini_restore('sendmail_from');
209:       $this->clearObject();
210:       return TRUE;
211:     }
212:   }

262:     // Check for abc@domain.com or Mister Smither <abc@domain.com>
263:     $regex = "/^[-0-9a-zA-Z.+_]+@[-0-9a-zA-Z.+_]+\.[a-zA-Z]{2,4}$|^[ a-zA-Z.+_]+<[-0-9a-zA-Z.+_]+@[-0-9a-zA-Z.+_]+\.[a-zA-Z]{2,4}>$/";
